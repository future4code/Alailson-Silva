function inverteArray(array) {

  const arrayInversa = []
  for(let i = array.length - 1; i >= 0; i--) {

    arrayInversa.push(array[i])
  }

  return arrayInversa
}

  let numeros = -Infinity, i = 0, n = array.length;

  for (let i = 0; i != n; ++i) {
      if (array[i] > numeros) {
          numeros = array[i];

  }

function checaTriangulo(a, b, c) {

  if(a === b && a === c) {
    return 'Equilátero'
  } else if((a === b && b !== c) || (a !== b && b === c)) {
    return 'Isósceles'
  } else if(a !== b && a !== c && b !== c) {
    return 'Escaleno'
  }

}
